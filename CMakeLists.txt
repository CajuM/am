project(am C)

	io/pulse.c
	encapsulation/enc1.c
	modulation/am.c
	modulation/amo.c
	modulation/fm.c

)

	./kiss_fft130/tools

	${PULSEAUDIO_LIBRARIES}

	kiss_fft
